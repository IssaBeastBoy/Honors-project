>>SQL code to create for each enzyme
CREATE TABLE `enzyme_variantinfo`.`EnzymeName` (
  `Variant number` INT NOT NULL AUTO_INCREMENT,
  `Variant ID` VARCHAR(45) NOT NULL,
  `Variant Name` VARCHAR(45) NOT NULL,
  `Variant Position` VARCHAR(45) NOT NULL,
  `Variant Effect` VARCHAR(100) NOT NULL,
  PRIMARY KEY (`Variant number`),
  UNIQUE INDEX `Variant number_UNIQUE` (`Variant number` ASC) VISIBLE,
  UNIQUE INDEX `Variant ID_UNIQUE` (`Variant ID` ASC) VISIBLE);

>>SQL code to create drugs that are affected
CREATE TABLE `enzyme_variantinfo`.`Name` (
  `Number` INT NOT NULL AUTO_INCREMENT,
  `Drug Name` VARCHAR(45) NOT NULL,
  `Enzyme` VARCHAR(100) NOT NULL,
  `Function` VARCHAR(200) NOT NULL,
  `Structure` VARCHAR(45) NOT NULL,
  `Metabolite(s)` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`Number`),
  UNIQUE INDEX `Number_UNIQUE` (`Number` ASC) VISIBLE,
  UNIQUE INDEX `Drug Name_UNIQUE` (`Drug Name` ASC) VISIBLE);
